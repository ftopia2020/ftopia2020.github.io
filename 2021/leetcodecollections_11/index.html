func spiralOrder(matrix [][]int) []int {
    if len(matrix)==0 {
        return []int{}
    }
    lenX := len(matrix[0])
    lenY := len(matrix)
    list := []int{}
    numLoop := 2*lenX
    if lenY<lenX {
        numLoop = 2*lenY
    }
    for n:=0; n<numLoop; n++ {
        fmt.Println(n)
        if n%4 == 0 { // â†’
            list = append(list, matrix[n/4][n/4:lenX-n/4]...)
            fmt.Println(list)
        }
        if n%4 == 1 { // â†“
            for i:=(n-1)/4+1; i< lenY-(n-1)/4; i++ {
                list = append(list, matrix[i][lenX-(n-1)/4-1])
            }
            fmt.Println(list)
        }
        if n%4 == 2 { // â†
            for j:=lenX-(n-2)/4-2; j>=(n-2)/4; j-- {
                list = append(list, matrix[lenY-(n-2)/4-1][j])
            }
            fmt.Println(list)
        }
        if n%4 == 3 { // â†‘
            for i:=lenY-(n-3)/4-2; i>(n-3)/4; i-- {
                list = append(list, matrix[i][(n-3)/4])
            }
            fmt.Println(list)
        }
    }
    return list
}
